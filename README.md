```bash
cd home-office-gear
git init
git add .
git commit -m "Initial site — 10 affiliate articles"
git remote add origin https://github.com/Demi-urge/home-office-gear.git
git push -u origin main
```

Then enable GitHub Pages in repo Settings → Pages → Source: main branch.
